directories=no, --> Brak katalogów
menubar=no, --> Brak paska menu
height=280, --> Wysokość=280
width=160, --> Szerokość=160
top=200, --> Oddalone o 200 od górnej krawędzi
left=200 --> Oddalone o 200 od lewej krawędzi
window.close() --> Zamyka okno
